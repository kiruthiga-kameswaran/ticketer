docker network create network-name
docker volume create volumename
run db container: 
docker run -d --name container-name --network network-name -e postgresuser,
password,dbname -v pgdata:/var/lib/postgresql/pgdata -p 5432:5432 postgres 
build app image: 
docker build -t next-app .
prisma migration: 
  docker run --rm 
  --network my-app-network 
  -e DATABASE_URL="postgresql://postgres:password@postgres-container:5432/mydb" 
  -v %cd%:/app 
  -w /app 
  node:18 
  sh -c "npm install && npx prisma migrate deploy"
run the app container: 
docker run -d --network network-name --name container-name -e DATABASE_URL="" -p 3000:3000 next-app
